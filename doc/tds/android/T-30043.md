### 公告弹出时机
```kotlin
onEnterRoom { room ->
    if (room.announcement.isNotBlank()) {
        val last = seenStore.get(room.id)
        if (last == null || now - last > 24.hours) {
            showPopup(room.announcement)
            seenStore.save(room.id, now)
        }
    }
}
```

### RoleBadge
```kotlin
@Composable
fun RoleBadge(role: Role, modifier: Modifier = Modifier) = when (role) {
```
