### AnalyticsPort 接口
```kotlin
interface AnalyticsPort {
    fun track(event: String, properties: Map<String, Any?> = emptyMap())
    fun setUser(userId: String?, traits: Map<String, Any?> = emptyMap())
    fun captureException(throwable: Throwable, extras: Map<String, Any?> = emptyMap())
    fun setConsent(mode: ConsentMode)  // All / CrashOnly / None
}

enum class ConsentMode { All, CrashOnly, None }
```

### SentryAnalytics 实现关键点
```kotlin
class SentryAnalytics @Inject constructor(
    @ApplicationContext ctx: Context,
    private val filter: SensitiveFilter
) : AnalyticsPort {
    init {
        SentryAndroid.init(ctx) { opts ->
            opts.dsn = BuildConfig.SENTRY_DSN
            opts.environment = BuildConfig.BUILD_TYPE // dev / staging / prod
            opts.beforeSend = BeforeSendCallback { evt, _ ->
                filter.scrub(evt)
                evt
            }
            opts.tracesSampleRate = 0.1
            opts.isEnableAnrDetection = true
            opts.isAttachThreads = true
        }
    }
    override fun captureException(t: Throwable, extras: Map<String, Any?>) {
        Sentry.captureException(filter.scrubThrowable(t)) { scope ->
            extras.forEach { (k, v) -> scope.setExtra(k, v?.toString() ?: "null") }
        }
    }
    // track/setUser：此类不负责业务事件上报，仅转发给 EventReportClient（T-30035）
    // MVP 暂让 SentryAnalytics 作为单一 Port，内部组合 EventReportClient
    // ...
}
```
