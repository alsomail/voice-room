#### `RoomPagingSource`

```kotlin
class RoomPagingSource(
    private val repository: IRoomRepository
) : PagingSource<Int, RoomItem>() {

    override fun getRefreshKey(state: PagingState<Int, RoomItem>): Int? =
        state.anchorPosition?.let { anchor ->
            state.closestPageToPosition(anchor)?.run {
                prevKey?.plus(1) ?: nextKey?.minus(1)
            }
        }

    override suspend fun load(params: LoadParams<Int>): LoadResult<Int, RoomItem> {
        val page = params.key ?: 1
        val size = params.loadSize.coerceAtMost(100)
        return repository.getRooms(page, size)
            .fold(
                onSuccess = { data ->
                    LoadResult.Page(
                        data = data.items,
                        prevKey = if (page > 1) page - 1 else null,
                        nextKey = if (data.total > page * size) page + 1 else null
                    )
                },
                onFailure = { error ->
                    LoadResult.Error(error)
                }
            )
    }
}
```

#### `RoomListViewModel`

```kotlin
class RoomListViewModel(
    private val repository: IRoomRepository = FakeRoomRepository()
) : ViewModel() {

    val pagingFlow: Flow<PagingData<RoomItem>> = Pager(
        config = PagingConfig(
            pageSize = 20,
            enablePlaceholders = false,
            prefetchDistance = 5
        )
    ) {
        repository.getRoomsPagingSource()
    }.flow.cachedIn(viewModelScope)

    class Factory(private val repository: IRoomRepository) : ViewModelProvider.Factory {
        @Suppress("UNCHECKED_CAST")
        override fun <T : ViewModel> create(modelClass: Class<T>): T =
            RoomListViewModel(repository) as T
    }
}
```

#### `HallScreen` 升级

```kotlin
@Composable
fun HallScreen(
    pagingItems: LazyPagingItems<RoomItem>,
    onNavigateToRoom: (String) -> Unit = {}
) {
    var isRefreshing by remember { mutableStateOf(false) }
    LaunchedEffect(pagingItems.loadState.refresh) {
        isRefreshing = pagingItems.loadState.refresh is LoadState.Loading
    }

    SwipeRefresh(
        state = rememberSwipeRefreshState(isRefreshing),
        onRefresh = { pagingItems.refresh() }
    ) {
        when (val refresh = pagingItems.loadState.refresh) {
            is LoadState.Loading -> CircularProgressIndicator(Modifier.testTag("hall_loading"))
            is LoadState.Error -> Column {
                Text(refresh.error.message ?: "加载失败", Modifier.testTag("hall_error_text"))
                Button(onClick = { pagingItems.retry() }, Modifier.testTag("hall_retry_button")) {
                    Text("重试")
                }
            }
            is LoadState.NotLoading -> {
                if (pagingItems.itemCount == 0) {
                    Text("暂无房间", Modifier.testTag("hall_empty_state"))
                } else {
                    LazyVerticalGrid(columns = GridCells.Adaptive(160.dp)) {
                        items(pagingItems.itemCount, key = { pagingItems.peek(it)?.roomId ?: it }) { idx ->
                            pagingItems[idx]?.let { room ->
                                RoomCard(room = room, onClick = { onNavigateToRoom(room.roomId) })
                            }
                        }
                        if (pagingItems.loadState.append is LoadState.Loading) {
                            item { CircularProgressIndicator(Modifier.testTag("hall_loading_more")) }
                        }
                    }
                }
            }
        }
    }
}
```

## 五、Reviewer意见

> 状态枚举：🟢 通过 | 🔴 未通过

### Round 1
- **状态**：🔴 未通过
- **审查意见**：发现 1 项 HIGH 级别问题阻塞合并。
- **改进建议**：
  1. **[HIGH] `PagingConfig` 缺少 `initialLoadSizeHint`（`RoomListViewModel.kt:32`）**  
     Paging3 默认 `initialLoadSizeHint = pageSize × 3 = 60`，首次 Refresh 以 `loadSize=60` 调用 `getRooms(page=1, size=60)` 返回 items[1..60]；后续 Append key=2 以 `loadSize=20` 调用 `getRooms(page=2, size=20)` 返回 items[21..40]，与首次已加载数据**重叠 40 条**，列表出现重复项。现有所有测试均人为传入 `loadSize=20` 给 `LoadParams.Refresh`，掩盖了此生产行为。  
     **修复**：在 `PagingConfig` 中显式设置 `initialLoadSizeHint = 20`（或 `= pageSize`），并新增 P01-extra 测试：`Refresh(loadSize=60)` 场景验证 nextKey 正确且与 Append 无数据重叠。

### Round 2
- **状态**：🟢 通过
- **审查意见**：HIGH 修复验证通过，`RoomListViewModel.kt:38` 已正确添加 `initialLoadSize = 20`，使用 Paging3 正确参数名（非废弃的 Paging2 `initialLoadSizeHint`），Refresh 与 Append 统一使用 loadSize=20，消除 items[1..60] ∩ items[21..40] 重叠问题。
- **MEDIUM 建议（不阻塞合并）**：
  1. **P01-extra 测试名语义相反**（`RoomPagingSourceTest.kt:203`）：测试名声称 "does not overlap Append"，但测试实际演示的是旧版 `loadSize=60` 的 BUG 场景（60条数据 + nextKey=2 确实会导致 Append 重叠）。建议将测试名改为 `"P01-extra documents overlap when loadSize=60 is used bug mitigated by initialLoadSize=20"`，或增加第二阶段断言验证 Append 数据不在已加载范围内。
  2. **V04 断言无效**（`RoomListViewModelTest.kt:164`）：仅 `assertNotNull(pagingFlow)`，与 V01 完全等价；若有人删除 `initialLoadSize = 20`，V04 依然通过。建议通过反射或内联暴露 config 的方式验证实际配置值，或将 V04 重命名为 "smoke test"。同时注释中 `initialLoadSizeHint` 系 Paging2 废弃术语，应统一改为 `initialLoadSize`。
