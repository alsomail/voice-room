### 表单 State
```kotlin
data class CreateRoomUiState(
    val title: String = "",
    val coverUrl: String = "",           // 从 T-30037 返回
    val category: RoomCategory = CHAT,   // 枚举
    val announcement: String = "",
    val passwordEnabled: Boolean = false,
    val password: String = "",           // 6 位数字
    val submitting: Boolean = false,
    val error: String? = null
) {
    val canSubmit: Boolean
        get() = title.isNotBlank() && title.length <= 30
             && announcement.length <= 200
             && (!passwordEnabled || password.matches(Regex("\\d{6}")))
             && coverUrl.isNotEmpty()
             && !submitting
}
```

### 提交
```kotlin
fun submit() {
    viewModelScope.launch {
        _state.update { it.copy(submitting = true) }
        val req = CreateRoomRequest(
            title = state.title,
            room_type = if (state.passwordEnabled) "password" else "normal",
            cover_url = state.coverUrl,
            category = state.category.key,
            announcement = state.announcement.ifBlank { null },
            password = state.password.ifBlank { null }
        )
        runCatching { roomApi.createRoom(req) }
            .onSuccess { navigateToRoom(it.room_id) }
            .onFailure { handleError(it) }
    }
}
```
